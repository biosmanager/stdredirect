210 static STDREDIRECT_ERROR STDREDIRECT_redirect(STDREDIRECT_REDIRECTION* redirection) {
211  /* unredirect if already redirected */
212  if (redirection->isRedirected && !STDREDIRECT_unredirect(redirection)) {
213  goto Error;
214  }
215 
216  /* create anonymous pipe */
217  if (!CreatePipe(&redirection->readablePipeEnd, &redirection->writablePipeEnd, 0, 0)) {
218  goto Error;
219  }
220 
222  //redirection->stdHandle = GetStdHandle(redirection->stream == STDREDIRECT_STREAM_STDOUT ? STD_OUTPUT_HANDLE : STD_ERROR_HANDLE);
223  //if (redirection->stdHandle == NULL) {
224  // goto Error;
225  //}
226 
227  /* set stream handle to writable pipe end */
228  if (!SetStdHandle(redirection->stream == STDREDIRECT_STREAM_STDOUT ? STD_OUTPUT_HANDLE : STD_ERROR_HANDLE, redirection->writablePipeEnd)) {
229  goto Error;
230  }
231 
232  /* assign OS handle to C-runtime file descriptor */
233  redirection->writablePipeEndFileDescriptor = _open_osfhandle((intptr_t) redirection->writablePipeEnd, 0);
234  if (redirection->writablePipeEndFileDescriptor == -1) {
235  goto Error;
236  }
237 
238  /* reassign standard stream file descriptor to writable pipe end */
239  if (_dup2(redirection->writablePipeEndFileDescriptor, redirection->stream == STDREDIRECT_STREAM_STDOUT ? _fileno(stdout) : _fileno(stderr)) == -1) {
240  goto Error;
241  }
242 
243  /* open handle to default console screen buffer */
244  redirection->consoleScreenBuffer = CreateFile("CONOUT$", GENERIC_WRITE, FILE_SHARE_WRITE, NULL, OPEN_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL);
245  if (redirection->consoleScreenBuffer == INVALID_HANDLE_VALUE) {
246  goto Error;
247  }
248 
249  /* create exit thread event */
250  redirection->exitThreadEvent = CreateEvent(NULL, TRUE, FALSE, NULL);
251  if (redirection->exitThreadEvent == NULL) {
252  goto Error;
253  }
254 
255  /* run pipe reader in separate thread */
256  redirection->thread = CreateThread(0, 0, (LPTHREAD_START_ROUTINE) STDREDIRECT_bufferedPipeReader, redirection, 0, 0);
257  if (redirection->thread == NULL) {
258  goto Error;
259  }
260 
261  redirection->isRedirected = TRUE;
262  redirection->isValid = TRUE;
263 
264  return redirection->error = STDREDIRECT_ERROR_NO_ERROR;
265 
266 Error:
267  /* cleanup */
268  STDREDIRECT_unredirect(redirection);
269 
270  return redirection->error = STDREDIRECT_ERROR_REDIRECT;
271 }

349 static STDREDIRECT_ERROR STDREDIRECT_unredirect(STDREDIRECT_REDIRECTION* redirection) {
350  FILE* consoleFile;
351 
352  /* stop pipe reader thread */
353  if (redirection->thread) {
354  /* signal thread to exit */
355  if (!SetEvent(redirection->exitThreadEvent)) {
356  goto Error;
357  }
358  /* check if thread exited itself, terminate otherwise */
359  if (WaitForSingleObject(redirection->thread, STDREDIRECT_THREAD_EXIT_TIMEOUT_MS) != WAIT_OBJECT_0 && !TerminateThread(redirection->thread, EXIT_SUCCESS)) {
360  goto Error;
361  }
362  /* check thread exit code */
363  DWORD threadExitCode;
364  if (!GetExitCodeThread(redirection->thread, &threadExitCode) || (threadExitCode && threadExitCode != EXIT_SUCCESS)) {
365  goto Error;
366  }
367 
368  /* close thread handle*/
369  if (!CloseHandle(redirection->thread)) {
370  goto Error;
371  }
372  redirection->thread = NULL;
373  }
374 
375  /* close exit thread event handle */
376  if (redirection->exitThreadEvent && !CloseHandle(redirection->exitThreadEvent)) {
377  goto Error;
378  }
379  redirection->exitThreadEvent = NULL;
380 
381 
382  /* free read buffer */
383  if (redirection->buffer) {
384  free(redirection->buffer);
385  redirection->buffer = NULL;
386  redirection->bufferSize = 0;
387  }
388 
390  //if (redirection->stdHandle && !SetStdHandle(redirection->stream == STDREDIRECT_STREAM_STDOUT ? STD_OUTPUT_HANDLE : STD_ERROR_HANDLE, redirection->stdHandle)) {
391  // goto Error;
392  //}
393  //redirection->stdHandle = NULL;
394 
395  /* close writable pipe end file descriptor */
396  if (redirection->writablePipeEndFileDescriptor != 1 && _close(redirection->writablePipeEndFileDescriptor) == -1) {
397  goto Error;
398  }
399  redirection->writablePipeEndFileDescriptor = -1;
400  redirection->writablePipeEnd = NULL;
401 
402  /* close readable pipe end */
403  if (redirection->readablePipeEnd && !CloseHandle(redirection->readablePipeEnd)) {
404  goto Error;
405  }
406  redirection->readablePipeEnd = NULL;
407 
408  /* re-open console */
409  if (freopen_s(&consoleFile, "CONOUT$", "w", redirection->stream == STDREDIRECT_STREAM_STDOUT ? stdout : stderr) != 0) {
410  goto Error;
411  }
412 
413  /* close handle to attached console that it used for output duplication */
414  if (!CloseHandle(redirection->consoleScreenBuffer)) {
415  goto Error;
416  }
417  redirection->consoleScreenBuffer = NULL;
418 
419  redirection->isRedirected = FALSE;
420  redirection->isValid = TRUE;
421 
422  return redirection->error = STDREDIRECT_ERROR_NO_ERROR;
423 
424 Error:
425  /* CRITICAL unredirect failed */
426 
427  /* try re-opening console output and show error message */
428  freopen_s(&consoleFile, "CONOUT$", "w", stdout);
429  printf("STDREDIRECT CRITICAL ERROR: Could not un-redirect %s! Redirection is in invalid state.\n", redirection->stream == STDREDIRECT_STREAM_STDOUT ? "stdout" : "stderr");
430 
431  redirection->isValid = FALSE;
432 
433  return redirection->error = STDREDIRECT_ERROR_UNREDIRECT;
434 }

462 static void WINAPI STDREDIRECT_bufferedPipeReader(STDREDIRECT_REDIRECTION* redirection) {
463  DWORD numBytesRead;
464  DWORD numBytesWritten;
465 
466  /* allocate string buffer */
467  redirection->buffer = (char*) calloc(STDREDIRECT_BUFFER_SIZE, sizeof(char));
468  if (redirection->buffer) {
469  redirection->bufferSize = STDREDIRECT_BUFFER_SIZE;
470  }
471  else {
472  goto Error;
473  }
474 
475  while (WaitForSingleObject(redirection->exitThreadEvent, 1) == WAIT_TIMEOUT) {
476  /* flush all streams so they become readable */
477  if (fflush(NULL) == EOF) {
478  goto Error;
479  }
480 
481  /* read from readable pipe end, blocks until input is available */
482  /* read 1 character less to leave space for string-terminating null-character */
483  if (!ReadFile(redirection->readablePipeEnd, (void*) redirection->buffer, (DWORD) redirection->bufferSize - 1, &numBytesRead, NULL)) {
484  goto Error;
485  }
486  if (numBytesRead > 0) {
487  /* ensure string is null-terminated */
488  redirection->buffer[redirection->bufferSize - 1] = '\0';
489 
490  /* duplicate output to attached console */
491  if (redirection->behaviour == STDREDIRECT_BEHAVIOUR_DUPLICATE) {
492  WriteConsole(redirection->consoleScreenBuffer, redirection->buffer, numBytesRead, &numBytesWritten, NULL);
493  }
494 
495  /* run string callback */
496  redirection->callback(redirection->buffer);
497  }
498 
499  /* clear buffer */
500  memset(redirection->buffer, 0, redirection->bufferSize);
501  }
502 
503  ExitThread(EXIT_SUCCESS);
504 
505 Error:
506  /* cleanup */
507 
508  /* close handle and set thread handle to null before unredirect so it doesn't try to terminate this thread */
509  CloseHandle(redirection->thread);
510  redirection->thread = NULL;
511  STDREDIRECT_unredirect(redirection);
512 
513  redirection->error = STDREDIRECT_ERROR_THREAD;
514 
515  ExitThread(EXIT_FAILURE);
516 }
